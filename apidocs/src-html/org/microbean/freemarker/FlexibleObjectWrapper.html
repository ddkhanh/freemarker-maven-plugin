017package org.microbean.freemarker;
018
019import java.util.Map;
020
021import freemarker.ext.beans.BeansWrapperConfiguration;
022
023import freemarker.ext.util.ModelFactory;
024
025import freemarker.template.DefaultObjectWrapper;
026import freemarker.template.DefaultObjectWrapperConfiguration;
027import freemarker.template.Version;
028
029/**
030 * A {@link DefaultObjectWrapper} that uses a {@link Map} of {@link
031 * ModelFactory} instances indexed by {@link Class} instances to
032 * assist in {@linkplain DefaultObjectWrapper#wrap(Object) wrapping
033 * <code>Object</code>s}.
034 *
035 * @author <a href="https://about.me/lairdnelson"
036 * target="_parent">Laird Nelson</a>
037 *
038 * @see #getModelFactory(Class)
039 */
040public class FlexibleObjectWrapper extends DefaultObjectWrapper {
041
042
043  /*
044   * Instance fields.
045   */
046
047
048  /**
049   * A {@link Map} of {@link ModelFactory} instances indexed by the
050   * {@link Class} instances for which they are suitable.
051   *
052   * <p>This field may be {@code null}.</p>
053   */
054  private final Map<? extends Class<?>, ? extends ModelFactory> modelFactories;
055
056
057  /*
058   * Constructors.
059   */
060
061
062  /**
063   * Creates a new {@link FlexibleObjectWrapper}.
064   *
065   * @param configuration the {@link BeansWrapperConfiguration}
066   * describing how this {@link FlexibleObjectWrapper} will be set up;
067   * must not be {@code null}
068   *
069   * @param writeProtected whether the resulting {@link
070   * FlexibleObjectWrapper} will be immutable from this point on
071   *
072   * @param modelFactories a {@link Map} of {@link
073   * ModelFactory} instances indexed by {@link Class} instances to
074   * assist in {@linkplain DefaultObjectWrapper#wrap(Object) wrapping
075   * <code>Object</code>s}; may be {@code null}
076   *
077   * @see
078   * DefaultObjectWrapper#DefaultObjectWrapper(BeansWrapperConfiguration,
079   * boolean)
080   */
081  public FlexibleObjectWrapper(final BeansWrapperConfiguration configuration,
082                               final boolean writeProtected,
083                               final Map<? extends Class<?>, ? extends ModelFactory> modelFactories) {
084    super(configuration, writeProtected);
085    this.modelFactories = modelFactories;
086  }
087
088  /**
089   * Creates a new {@link FlexibleObjectWrapper}.
090   *
091   * @param configuration the {@link DefaultObjectWrapperConfiguration}
092   * describing how this {@link FlexibleObjectWrapper} will be set up;
093   * must not be {@code null}
094   *
095   * @param writeProtected whether the resulting {@link
096   * FlexibleObjectWrapper} will be immutable from this point on
097   *
098   * @param modelFactories a {@link Map} of {@link
099   * ModelFactory} instances indexed by {@link Class} instances to
100   * assist in {@linkplain DefaultObjectWrapper#wrap(Object) wrapping
101   * <code>Object</code>s}; may be {@code null}
102   *
103   * @see
104   * DefaultObjectWrapper#DefaultObjectWrapper(BeansWrapperConfiguration,
105   * boolean)
106   */
107  public FlexibleObjectWrapper(final DefaultObjectWrapperConfiguration configuration,
108                               final boolean writeProtected,
109                               final Map<? extends Class<?>, ? extends ModelFactory> modelFactories) {
110    super(configuration, writeProtected);
111    this.modelFactories = modelFactories;
112  }
113  
114  /**
115   * Creates a new {@link FlexibleObjectWrapper}.
116   *
117   * @param version the {@link Version} representing the Freemarker
118   * version in use; must not be {@code null}
119   *
120   * @param modelFactories a {@link Map} of {@link
121   * ModelFactory} instances indexed by {@link Class} instances to
122   * assist in {@linkplain DefaultObjectWrapper#wrap(Object) wrapping
123   * <code>Object</code>s}; may be {@code null}
124   *
125   * @see
126   * DefaultObjectWrapper#DefaultObjectWrapper(BeansWrapperConfiguration,
127   * boolean)
128   */
129  public FlexibleObjectWrapper(final Version version,
130                               final Map<? extends Class<?>, ? extends ModelFactory> modelFactories) {
131    super(version);
132    this.modelFactories = modelFactories;
133  }
134
135
136  /*
137   * Instance methods.
138   */
139
140
141  /**
142   * Returns a {@link ModelFactory} suitable for the supplied {@link
143   * Class}.
144   *
145   * <p>This method never returns {@code null}.</p>
146   *
147   * <p>Overrides of this method must never return {@code null}.</p>
148   *
149   * <p>This implementation tries to return a {@link ModelFactory} as
150   * found in the {@linkplain #FlexibleObjectWrapper(Version, Map)
151   * <code>Map</code> of such <code>ModelFactory</code> instances
152   * provided at construction time}.  If there is no such {@link
153   * ModelFactory}, then the return value of invoking the {@link
154   * DefaultObjectWrapper#getModelFactory(Class)
155   * super.getModelFactory(Class)} is returned instead.</p>
156   *
157   * @param c the {@link Class} for which a {@link ModelFactory} is
158   * desired; must not be {@code null}
159   *
160   * @return a {@link ModelFactory}; never {@code null}
161   *
162   * @see DefaultObjectWrapper#getModelFactory(Class)
163   */
164  @Override
165  @SuppressWarnings("rawtypes")
166  protected ModelFactory getModelFactory(final Class c) {
167    ModelFactory returnValue = null;
168    if (c != null && this.modelFactories != null && !this.modelFactories.isEmpty()) {
169      returnValue = this.modelFactories.get(c);
170    }
171    if (returnValue == null) {
172      returnValue = super.getModelFactory(c);
173    }
174    return returnValue;
175
176  }
177  
178}
