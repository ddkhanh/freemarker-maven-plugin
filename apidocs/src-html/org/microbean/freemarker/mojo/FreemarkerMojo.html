317  /**
318   * Executes the {@code freemarker} goal.
319   *
320   * @exception MojoExecutionException if a fatal error occured
321   */
322  @Override
323  public void execute() throws MojoExecutionException {
324    final Log log = this.getLog();
325    
326    if (this.isSkip()) {
327      if (log != null && log.isInfoEnabled()) {
328        log.info("Skipping execution by request.");
329      }
330      return;
331    }
332    
333    //
334    // Determine ModelFactory instances in use.
335    //
336    
337    Map<String, ModelFactory> modelFactoriesByClassName = this.getModelFactories();
338    if (modelFactoriesByClassName == null) {
339      modelFactoriesByClassName = new HashMap<>();
340    }
341    modelFactoriesByClassName.putIfAbsent("java.lang.Class", new ClassModelFactory());
342    
343    Map<? extends Class<?>, ? extends ModelFactory> modelFactories = null;
344    try {
345      modelFactories = convert(modelFactoriesByClassName);
346    } catch (final ClassNotFoundException classNotFoundException) {
347      throw new MojoExecutionException(classNotFoundException.getMessage(), classNotFoundException);
348    }
349    assert modelFactories != null;
350
351    if (log != null && log.isDebugEnabled()) {
352      log.debug("Using modelFactories: " + modelFactories);
353    }
354
355    //
356    // Determine Configuration in use.
357    //
358    
359    Configuration configuration = this.getConfiguration();
360    if (configuration == null) {
361      configuration = new Configuration(FREEMARKER_VERSION);
362      configuration.setDefaultEncoding("UTF-8");
363      configuration.setAPIBuiltinEnabled(true);
364    }
365    configuration.setIncompatibleImprovements(FREEMARKER_VERSION);
366    final DefaultObjectWrapper beansWrapper = new FlexibleObjectWrapper(FREEMARKER_VERSION, modelFactories);
367    beansWrapper.setExposureLevel(BeansWrapper.EXPOSE_ALL); // http://freemarker.org/docs/api/freemarker/ext/beans/BeansWrapper.html#setExposureLevel-int-
368    beansWrapper.setForceLegacyNonListCollections(false); // http://freemarker.org/docs/api/freemarker/template/DefaultObjectWrapper.html#setForceLegacyNonListCollections-boolean-
369    beansWrapper.setUseAdaptersForContainers(true); // http://freemarker.org/docs/api/freemarker/template/DefaultObjectWrapper.html#setUseAdaptersForContainers-boolean-
370    configuration.setObjectWrapper(beansWrapper);
371    assert beansWrapper == configuration.getObjectWrapper();
372    configuration.setTemplateExceptionHandler(TemplateExceptionHandler.RETHROW_HANDLER);
373    configuration.setLogTemplateExceptions(false);
374    if (log != null && log.isDebugEnabled()) {
375      log.debug("Using configuration: " + configuration);
376    }
377    
378    //
379    // Determine what template names to use.  There may be just one,
380    // and there must be at least one.
381    //
382    
383    final Set<String> templateNames = this.getTemplateNames(configuration);
384    if (templateNames == null || templateNames.isEmpty()) {
385      throw new MojoExecutionException("No templates to process");
386    } else if (templateNames.size() > 1) {
387      // If there are lots of templates to process, then treat
388      // outputFile as a directory, not a regular file.
389      final File outputFile = this.getOutputFile();
390      if (outputFile != null) {
391        if (outputFile.exists() && !outputFile.isDirectory()) {
392          throw new MojoExecutionException("outputFile was an existing non-directory: " + outputFile);
393        } else {
394          outputFile.mkdirs();
395        }
396      }
397    }
398    if (log != null && log.isDebugEnabled()) {
399      log.debug("Using templateNames: " + templateNames);
400    }
401
402
403    //
404    // Find the Jandex index if we can get it.
405    //
406    
407    IndexView indexView = null;
408    File jandexIndexFile = this.getJandexIndexFile();
409    if (jandexIndexFile != null && jandexIndexFile.isFile() && jandexIndexFile.canRead()) {
410      IndexReader reader = null;
411      try (final InputStream inputStream = new BufferedInputStream(new FileInputStream(jandexIndexFile))) {
412        indexView = new IndexReader(inputStream).read();
413      } catch (final IOException ioException) {
414        throw new MojoExecutionException(ioException.getMessage(), ioException);
415      }
416      assert indexView != null;
417    } else {
418      // TODO: index on the fly
419    }
420    if (log != null && log.isDebugEnabled()) {
421      log.debug("Using Jandex index file: " + jandexIndexFile);
422    }
423
424    //
425    // Set up the data model for the Freemarker template engine.
426    //
427    
428    Map<String, Object> dataModel = this.getDataModel();
429    if (dataModel == null) {
430      dataModel = new HashMap<>();
431    }
432    dataModel.put("enums", beansWrapper.getEnumModels()); // note: not putIfAbsent
433    dataModel.put("statics", beansWrapper.getStaticModels()); // note: not putIfAbsent
434    dataModel.putIfAbsent("classIndex", indexView);
435    dataModel.putIfAbsent("project", this.getProject());
436    dataModel.putIfAbsent("session", this.getSession());
437    final ClassLoader contextClassLoader = Thread.currentThread().getContextClassLoader();
438    dataModel.putIfAbsent("classes", new LoadableClassesTemplateHashModel(beansWrapper, contextClassLoader));
439    dataModel.putIfAbsent("contextClassLoader", contextClassLoader);
440    if (log != null && log.isDebugEnabled()) {
441      log.debug("Using dataModel: " + dataModel);
442    }
443
444    //
445    // Determine what output encoding to use.
446    //
447
448    String outputEncoding = this.getOutputEncoding();
449    if (outputEncoding == null) {
450      outputEncoding = "UTF-8";
451    }
452    if (log != null && log.isDebugEnabled()) {
453      log.debug("Using outputEncoding: " + outputEncoding);
454    }
455    
456    //
457    // Process templates.
458    //
459    
460    for (final String templateName : templateNames) {
461      if (templateName == null) {
462        throw new MojoExecutionException("Encountered null template name in templateNames; check the getTemplateNames() method");
463      }
464
465      final File outputFile = this.getOutputFile(templateName);
466      if (outputFile == null) {
467        if (log != null && log.isDebugEnabled()) {
468          log.debug("No suitable outputFile found for a template with the name " + templateName + "; skipping processing");
469        }   
470      } else {
471        if (log != null && log.isDebugEnabled()) {
472          log.debug("Loading template: " + templateName);
473          log.debug("Output file: " + outputFile);
474        }
475        Template template = null;
476        try {
477          template = configuration.getTemplate(templateName);
478        } catch (final IOException exception) {
479          throw new MojoExecutionException(exception.getMessage(), exception);
480        }
481        if (log != null && log.isDebugEnabled()) {
482          log.debug("Compiled template: " + template);
483          log.debug("Processing...");
484        }        
485        try (final Writer out = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(outputFile), outputEncoding))) {
486          template.process(dataModel, out);
487          if (log != null && log.isDebugEnabled()) {
488            log.debug("...processing complete.");
489          }
490        } catch (final IOException | TemplateException exception) {
491          throw new MojoExecutionException(exception.getMessage(), exception);
492        }
493      }
494    }
495
496    if (log != null && log.isDebugEnabled()) {
497      log.debug("All template processing complete.");
498    }
499  }

501  /**
502   * Returns a {@link File} representing the path where the output of
503   * processing a Freemarker template with the supplied {@code
504   * templateName} should be deposited.
505   *
506   * <p>This method never returns {@code null}.</p>
507   *
508   * <p>Overrides of this method must not return {@code null}.</p>
509   *
510   * @param templateName the name of a Freemarker template; must not
511   * be {@code null}
512   *
513   * @return a non-{@code null} {@link File} representing the path
514   * where the output of processing a Freemarker template with the
515   * supplied {@code templateName} should be deposited
516   *
517   * @exception NullPointerException if {@code templateName} is {@code
518   * null}
519   *
520   * @see #getDefaultOutputFile(String)
521   *
522   * @see #getOutputFile()
523   */
524  protected File getOutputFile(final String templateName) {
525    Objects.requireNonNull(templateName);
526    File returnValue = null;
527    if (templateName.equals(this.getTemplateName())) {
528      // If we are being asked for the outputFile for the
529      // user-specified templateName, then use the user-specified
530      // outputFile if there is one.
531      returnValue = this.getOutputFile();
532    } else {
533      File outputFile = this.getOutputFile();
534      if (outputFile == null) {
535        outputFile = this.getProjectBuildDirectory();
536      }
537      assert outputFile != null;
538      returnValue = new File(outputFile, stripFtlSuffix(templateName));
539    }
540    if (returnValue == null) {
541      returnValue = this.getDefaultOutputFile(templateName);
542    }
543    assert returnValue != null;
544    final File directory = returnValue.getParentFile();
545    assert directory != null;
546    directory.mkdirs();
547    return returnValue;
548  }

550  /**
551   * Returns a {@link File} representing the <em>default location</em> for the
552   * output from the processing of a Freemarker template with the
553   * supplied {@code templateName}.
554   *
555   * <p>This method never returns {@code null}.</p>
556   *
557   * <p>Overrides of this method must not return {@code null}.</p>
558   *
559   * <p>This implementation returns a new {@link File} formed from a
560   * concatenation of the {@linkplain #getProjectBuildDirectory()
561   * default Maven project build directory} and the supplied {@code
562   * templateName}, minus its {@code .ftl} suffix (if it has one).</p>
563   *
564   * @param templateName the name of the template for which an output
565   * file should be returned; must not be {@code null}
566   *
567   * @return a non-{@code null} {@link File} representing the
568   * <em>default location</em> for the output from the processing of a
569   * Freemarker template with the supplied {@code templateName}
570   *
571   * @exception NullPointerException if {@code templateName} is {@code
572   * null}
573   *
574   * @see #getProjectBuildDirectory()
575   *
576   * @see #getOutputFile(String)
577   */
578  protected File getDefaultOutputFile(final String templateName) {
579    Objects.requireNonNull(templateName);
580    return new File(this.getProjectBuildDirectory(), stripFtlSuffix(templateName));
581  }

583  /**
584   * Removes any trailing "{@code .ftl}" from the supplied {@link
585   * String} and returns the result.
586   *
587   * <p>This method may return {@code null}.</p>
588   *
589   * @param templateName the name of the template whose optional
590   * {@code .ftl} suffix will be stripped; may be {@code null} in
591   * which case {@code null} will be returned
592   *
593   * @return the doctored template name, or {@code null}
594   */
595  private static final String stripFtlSuffix(final String templateName) {
596    String returnValue = templateName;
597    if (templateName != null) {
598      final int length = templateName.length();
599      // 4 == ".ftl".length()
600      if (length > 4 && templateName.toLowerCase().endsWith(".ftl")) {
601        returnValue = templateName.substring(0, length - 4);
602      }
603    }
604    return returnValue;
605  }

607  /**
608   * A convenience method that returns a {@link File} representing the
609   * {@linkplain Build#getDirectory() default Maven build directory}.
610   *
611   * <p>This method never returns {@code null}.</p>
612   *
613   * <p>The {@link File} returned by this method is guaranteed to
614   * {@linkplain File#isDirectory() be a directory}.</p>
615   *
616   * @return a non-{@code null} {@linkplain File#isDirectory()
617   * <code>File</code> representing a directory} that is the
618   * {@linkplain Build#getDirectory() default Maven build directory}
619   *
620   * @see Build#getDirectory()
621   */
622  protected final File getProjectBuildDirectory() {
623    final MavenProject project = this.getProject();
624    if (project == null) {
625      throw new IllegalStateException("project cannot be null");
626    }
627    final Build projectBuild = project.getBuild();
628    if (projectBuild == null) {
629      throw new IllegalStateException("project.getBuild() cannot return null");
630    }
631    final String projectBuildDirectory = projectBuild.getDirectory();
632    if (projectBuildDirectory == null) {
633      throw new IllegalStateException("project.getBuild().getDirectory() cannot return null");
634    }
635    final File projectBuildDirectoryFile = new File(projectBuildDirectory);
636    assert projectBuildDirectoryFile.isDirectory();
637    return projectBuildDirectoryFile;
638  }

836  /**
837   * Returns a {@link Set} of names of Freemarker templates to
838   * process.
839   *
840   * <p>This method never returns {@code null}.</p>
841   *
842   * <p>Overrides of this method must not return {@code null}.</p>
843   *
844   * @param configuration a {@link Configuration} that might be
845   * inspected for its {@link Configuration#getTemplateLoader()
846   * TemplateLoader}; may be {@code null}
847   *
848   * @return a non-{@code null}, {@linkplain
849   * Collections#unmodifiableSet(Set) unmodifiable <code>Set</code>}
850   * of template names
851   *
852   * @see #getTemplateName()
853   */
854  protected Set<String> getTemplateNames(final Configuration configuration) {
855    final Set<String> templateNames = new LinkedHashSet<>();
856    final String templateName = this.getTemplateName();
857    if (templateName == null) {
858      if (configuration != null) {
859        final TemplateLoader templateLoader = configuration.getTemplateLoader();
860        if (templateLoader instanceof FileTemplateLoader) {
861          final File templateLoaderDirectory = ((FileTemplateLoader)templateLoader).getBaseDirectory();
862          if (templateLoaderDirectory != null && templateLoaderDirectory.isDirectory()) {
863            final String[] ftlFilenames = templateLoaderDirectory.list((d, n) -> n != null && n.toLowerCase().endsWith(".ftl"));
864            if (ftlFilenames != null && ftlFilenames.length > 0) {
865              for (final String ftlFilename : ftlFilenames) {
866                if (ftlFilename != null) {
867                  templateNames.add(ftlFilename);
868                }
869              }
870            }
871          }
872        }
873      }
874    } else {
875      templateNames.add(templateName);
876    }
877    if (templateNames == null || templateNames.isEmpty()) {
878      return Collections.emptySet();
879    } else {
880      return Collections.unmodifiableSet(templateNames);
881    }
882  }

971  /**
972   * Converts a {@link Map} of classnames indexing {@link
973   * ModelFactory} instances into a {@link Map} of {@link Class}
974   * instances indexing the same {@link ModelFactory} instances by
975   * {@linkplain ClassLoader#loadClass(String) loading the
976   * <code>Class</code>} corresponding to each classname using the
977   * {@link Thread#getContextClassLoader() context classloader} and
978   * returns the result.
979   *
980   * <p>This method never returns {@code null}.</p>
981   *
982   * @param originalModelFactoryMap the {@link Map} to convert; may be
983   * {@code null} in which case an {@linkplain Map#isEmpty() empty}
984   * {@link Map} will be returned
985   *
986   * @return a non-{@code null} {@link Map} of {@link Class} instances
987   * indexing appropriate {@link ModelFactory} instances
988   *
989   * @exception ClassNotFoundException if a given {@link Class} could
990   * not be {@linkplain ClassLoader#loadClass(String) loaded}
991   *
992   * @see ClassLoader#loadClass(String)
993   *
994   * @see Thread#getContextClassLoader()
995   */
996  private static final Map<? extends Class<?>, ? extends ModelFactory> convert(final Map<? extends String, ? extends ModelFactory> originalModelFactoryMap) throws ClassNotFoundException {
997    final Map<Class<?>, ModelFactory> returnValue = new HashMap<>();
998    final ClassLoader cl = Thread.currentThread().getContextClassLoader();
999    assert cl != null;
1000    if (originalModelFactoryMap != null && !originalModelFactoryMap.isEmpty()) {
1001      final Iterable<? extends Entry<? extends String, ? extends ModelFactory>> entries = originalModelFactoryMap.entrySet();
1002      if (entries != null) {
1003        for (final Entry<? extends String, ? extends ModelFactory> entry : entries) {
1004          if (entry != null) {
1005            final String className = entry.getKey();
1006            if (className == null) {
1007              returnValue.put(null, entry.getValue());
1008            } else {
1009              returnValue.put(cl.loadClass(className), entry.getValue());
1010            }
1011          }
1012        }
1013      }
1014    }
1015    return returnValue;
1016  }
